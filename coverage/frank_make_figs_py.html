82def make_deprojection_fig(u, v, vis, geom, force_style=True, 

83 save_prefix=None): 

84 r""" 

85 Produce a simple figure showing the effect of deprojection on the (u, v) 

86 coordinates and visibilities 

87 

88 Parameters 

89 ---------- 

90 u, v : array, unit = :math:`\lambda` 

91 Projected (u, v) coordinates 

92 vis : array, unit = Jy 

93 Projected visibilities (complex: real + imag * 1j) 

94 geom : SourceGeometry object 

95 Fitted geometry (see frank.geometry.SourceGeometry) 

96 force_style: bool, default = True 

97 Whether to use preconfigured matplotlib rcParams in generated figure 

98 save_prefix : string, default = None 

99 Prefix for saved figure name. If None, the figure won't be saved 

100 

101 Returns 

102 ------- 

103 fig : Matplotlib `.Figure` instance 

104 The produced figure, including the GridSpec 

105 axes : Matplotlib `~.axes.Axes` class 

106 The axes of the produced figure 

107 """ 

108 

109 # Apply the deprojection to the provided (u, v) coordinates 

110 # and visibility amplitudes 

111 up, vp, visp = geom.apply_correction(u, v, vis) 

112 

113 re_vis = np.real(vis) 

114 re_visp = np.real(visp) 

115 

116 with frank_plotting_style_context_manager(force_style): 

117 gs = GridSpec(2, 1) 

118 fig = plt.figure(figsize=(8, 6)) 

119 

120 ax0 = fig.add_subplot(gs[0]) 

121 ax1 = fig.add_subplot(gs[1]) 

122 

123 axes = [ax0, ax1] 

124 

125 plot_deprojection_effect(u / 1e6, v / 1e6, up / 1e6, vp / 1e6, re_vis * 1e3, 

126 re_visp * 1e3, ax0, ax1) 

127 

128 ax0.set_xlabel(r'u [M$\lambda$]') 

129 ax0.set_ylabel(r'v [M$\lambda$]') 

130 ax1.set_xlabel(r'Baseline [M$\lambda$]') 

131 ax1.set_ylabel('Re(V) [mJy]') 

132 ax1.set_xscale('log') 

133 ax1.set_yscale('log') 

134 ax1.set_ylim(bottom=1e-3) 

135 

136 ax0.legend(loc=0) 

137 ax1.legend(loc=0) 

138 

139 plt.tight_layout() 

140 

141 if save_prefix: 

142 plt.savefig(save_prefix + '_frank_deprojection.png', dpi=600) 

143 plt.close() 

144 

145 return fig, axes

148def make_quick_fig(u, v, vis, weights, sol, bin_widths, dist=None, 

149 force_style=True, save_prefix=None, norm_residuals=False, 

150 ): 

151 r""" 

152 Produce a simple figure showing just a Frankenstein fit, not any diagnostics 

153 

154 Parameters 

155 ---------- 

156 u, v : array, unit = :math:`\lambda` 

157 u and v coordinates of observations 

158 vis : array, unit = Jy 

159 Observed visibilities (complex: real + imag * 1j) 

160 weights : array, unit = Jy^-2 

161 Weights assigned to observed visibilities, of the form 

162 :math:`1 / \sigma^2` 

163 sol : _HankelRegressor object 

164 Reconstructed profile using Maximum a posteriori power spectrum 

165 (see frank.radial_fitters.FrankFitter) 

166 bin_widths : list, unit = \lambda 

167 Bin widths in which to bin the observed visibilities 

168 dist : float, optional, unit = AU, default = None 

169 Distance to source, used to show second x-axis in [AU] 

170 force_style: bool, default = True 

171 Whether to use preconfigured matplotlib rcParams in generated figure 

172 save_prefix : string, default = None 

173 Prefix for saved figure name. If None, the figure won't be saved 

174 norm_residuals : bool, default = False 

175 Whether to normalize the residual visibilities by the data's 

176 visibility amplitudes 

177 

178 Returns 

179 ------- 

180 fig : Matplotlib `.Figure` instance 

181 The produced figure, including the GridSpec 

182 axes : Matplotlib `~.axes.Axes` class 

183 The axes of the produced figure 

184 """ 

185 

186 logging.info(' Making quick figure') 

187 

188 with frank_plotting_style_context_manager(force_style): 

189 gs = GridSpec(2, 2, hspace=0) 

190 fig = plt.figure(figsize=(8, 6)) 

191 

192 ax0 = fig.add_subplot(gs[0]) 

193 ax1 = fig.add_subplot(gs[2]) 

194 

195 ax2 = fig.add_subplot(gs[1]) 

196 ax3 = fig.add_subplot(gs[3]) 

197 

198 ax0.text(.5, .9, 'a)', transform=ax0.transAxes) 

199 ax1.text(.5, .9, 'b)', transform=ax1.transAxes) 

200 

201 ax2.text(.5, .9, 'c)', transform=ax2.transAxes) 

202 ax3.text(.5, .9, 'd)', transform=ax3.transAxes) 

203 

204 axes = [ax0, ax1, ax2, ax3] 

205 

206 total_flux = trapz(sol.mean * 2 * np.pi * sol.r, sol.r) 

207 plot_brightness_profile(sol.r, sol.mean / 1e10, ax0, c='r', 

208 label='frank, total flux {:.2e} Jy'.format(total_flux)) 

209 if dist: 

210 ax0_5 = plot_brightness_profile(sol.r, sol.mean / 1e10, ax0, dist=dist, c='r') 

211 xlims = ax0.get_xlim() 

212 ax0_5.set_xlim(np.multiply(xlims, dist)) 

213 

214 plot_brightness_profile(sol.r, sol.mean / 1e10, ax1, c='r', label='frank') 

215 

216 u_deproj, v_deproj, vis_deproj = sol.geometry.apply_correction(u, v, vis) 

217 baselines = (u_deproj**2 + v_deproj**2)**.5 

218 grid = np.logspace(np.log10(min(baselines.min(), sol.q[0])), 

219 np.log10(max(baselines.max(), sol.q[-1])), 

220 10**4) 

221 

222 for i in range(len(bin_widths)): 

223 binned_vis = UVDataBinner( 

224 baselines, vis_deproj, weights, bin_widths[i]) 

225 vis_re_kl = binned_vis.V.real * 1e3 

226 vis_err_re_kl = binned_vis.error.real * 1e3 

227 vis_fit = sol.predict_deprojected(binned_vis.uv).real * 1e3 

228 

229 resid = vis_re_kl - vis_fit 

230 if norm_residuals: 

231 resid /= vis_re_kl 

232 rmse = (np.mean(resid**2))**.5 

233 

234 plot_vis_quantity(binned_vis.uv, vis_re_kl, ax2, c=cs[i], 

235 marker=ms[i], ls='None', 

236 label=r'Obs., {:.0f} k$\lambda$ bins'.format(bin_widths[i]/1e3)) 

237 

238 plot_vis_quantity(binned_vis.uv, resid, ax3, c=cs[i], marker=ms[i], 

239 ls='None', 

240 label=r'{:.0f} k$\lambda$ bins, RMSE {:.3f} mJy'.format(bin_widths[i]/1e3, rmse)) 

241 

242 vis_fit_kl = sol.predict_deprojected(grid).real * 1e3 

243 plot_vis_quantity(grid, vis_fit_kl, ax2, c='r', label='frank') 

244 

245 ax1.set_xlabel('r ["]') 

246 ax0.set_ylabel(r'Brightness [$10^{10}$ Jy sr$^{-1}$]') 

247 ax1.set_ylabel(r'Brightness [$10^{10}$ Jy sr$^{-1}$]') 

248 ax1.set_yscale('log') 

249 ax1.set_ylim(bottom=1e-3) 

250 

251 ax3.set_xlabel(r'Baseline [$\lambda$]') 

252 ax2.set_ylabel('Re(V) [mJy]') 

253 if norm_residuals: 

254 ax3.set_ylabel('Norm. residual') 

255 else: 

256 ax3.set_ylabel('Residual [mJy]') 

257 ax2.set_xscale('log') 

258 ax3.set_xscale('log') 

259 

260 xlims = ax2.get_xlim() 

261 ax3.set_xlim(xlims) 

262 

263 plt.setp(ax0.get_xticklabels(), visible=False) 

264 plt.setp(ax2.get_xticklabels(), visible=False) 

265 

266 plt.tight_layout() 

267 

268 if save_prefix: 

269 plt.savefig(save_prefix + '_frank_fit_quick.png', dpi=600) 

270 plt.close() 

271 

272 return fig, axes

275def make_full_fig(u, v, vis, weights, sol, bin_widths, alpha, wsmooth, 

276 gamma=1.0, dist=None, force_style=True, save_prefix=None, 

277 norm_residuals=False): 

278 r""" 

279 Produce a figure showing a Frankenstein fit and some useful diagnostics 

280 

281 Parameters 

282 ---------- 

283 u, v : array, unit = :math:`\lambda` 

284 u and v coordinates of observations 

285 vis : array, unit = Jy 

286 Observed visibilities (complex: real + imag * 1j) 

287 weights : array, unit = Jy^-2 

288 Weights assigned to observed visibilities, of the form 

289 :math:`1 / \sigma^2` 

290 sol : _HankelRegressor object 

291 Reconstructed profile using Maximum a posteriori power spectrum 

292 (see frank.radial_fitters.FrankFitter) 

293 bin_widths : list, unit = \lambda 

294 Bin widths in which to bin the observed visibilities 

295 alpha : float 

296 Value for the :math:`\alpha` hyperparameter. 

297 Used for the plot legends 

298 wsmooth : float 

299 Value for the :math:`w_{smooth}` hyperparameter. 

300 Used for the plot legends 

301 gamma : float, default = 1.0 

302 Index of power law normalization to apply to swept profile image's 

303 colormap (see matplotlib.colors.PowerNorm). 

304 gamma=1.0 yields a linear colormap 

305 dist : float, optional, unit = AU, default = None 

306 Distance to source, used to show second x-axis in [AU] 

307 force_style: bool, default = True 

308 Whether to use preconfigured matplotlib rcParams in generated figure 

309 save_prefix : string, default = None 

310 Prefix for saved figure name. If None, the figure won't be saved 

311 norm_residuals : bool, default = False 

312 Whether to normalize the residual visibilities by the data's 

313 visibility amplitudes 

314 

315 Returns 

316 ------- 

317 fig : Matplotlib `.Figure` instance 

318 The produced figure, including the GridSpec 

319 axes : Matplotlib `~.axes.Axes` class 

320 The axes of the produced figure 

321 """ 

322 

323 logging.info(' Making full figure') 

324 

325 with frank_plotting_style_context_manager(force_style): 

326 gs = GridSpec(3, 3, hspace=0) 

327 gs1 = GridSpec(4, 3, hspace=0, top=.88) 

328 gs2 = GridSpec(3, 3, hspace=.35, left=.04) 

329 fig = plt.figure(figsize=(8, 6)) 

330 

331 ax0 = fig.add_subplot(gs[0]) 

332 ax1 = fig.add_subplot(gs[3]) 

333 ax2 = fig.add_subplot(gs2[6]) 

334 

335 ax3 = fig.add_subplot(gs[1]) 

336 ax4 = fig.add_subplot(gs[4]) 

337 ax5 = fig.add_subplot(gs[7]) 

338 

339 ax6 = fig.add_subplot(gs[2]) 

340 ax7 = fig.add_subplot(gs1[5]) 

341 ax8 = fig.add_subplot(gs1[8]) 

342 ax9 = fig.add_subplot(gs1[11]) 

343 

344 ax0.text(.9, .6, 'a)', transform=ax0.transAxes) 

345 ax1.text(.9, .6, 'b)', transform=ax1.transAxes) 

346 ax2.text(.1, .9, 'c)', c='w', transform=ax2.transAxes) 

347 

348 ax3.text(.1, .5, 'd)', transform=ax3.transAxes) 

349 ax4.text(.1, .7, 'e)', transform=ax4.transAxes) 

350 ax5.text(.1, .7, 'f)', transform=ax5.transAxes) 

351 ax6.text(.9, .9, 'g)', transform=ax6.transAxes) 

352 ax7.text(.9, .9, 'h)', transform=ax7.transAxes) 

353 ax8.text(.9, .9, 'i)', transform=ax8.transAxes) 

354 ax9.text(.9, .9, 'j)', transform=ax9.transAxes) 

355 

356 axes = [ax0, ax1, ax2, ax3, ax4, ax5, ax6, ax7, ax8, ax9] 

357 

358 # Calculate the fit's total flux (2D, by sweeping the 1D profile over 2\pi) 

359 total_flux = trapz(sol.mean * 2 * np.pi * sol.r, sol.r) 

360 

361 # Plot the fitted brightness profile in linear- and log-y 

362 plot_brightness_profile(sol.r, sol.mean / 1e10, ax0, c='r', 

363 label='frank, total flux {:.2e} Jy'.format(total_flux)) 

364 if dist: 

365 ax0_5 = plot_brightness_profile(sol.r, sol.mean / 1e10, ax0, dist=dist, c='r') 

366 xlims = ax0.get_xlim() 

367 ax0_5.set_xlim(np.multiply(xlims, dist)) 

368 

369 plot_brightness_profile(sol.r, sol.mean / 1e10, ax1, c='r', label='frank') 

370 

371 # Apply deprojection to the provided (u, v) coordinates 

372 # and visibility amplitudes 

373 u_deproj, v_deproj, vis_deproj = sol.geometry.apply_correction(u, v, vis) 

374 baselines = (u_deproj**2 + v_deproj**2)**.5 

375 # Set a grid of baselines on which to plot the visibility domain frank fit 

376 grid = np.logspace(np.log10(min(baselines.min(), sol.q[0])), 

377 np.log10(max(baselines.max(), sol.q[-1])), 10**4 

378 ) 

379 # Map the frank visibility fit to `grid`, considering only the real component 

380 # that frank fits 

381 vis_fit_kl = sol.predict_deprojected(grid).real * 1e3 

382 

383 # Make a guess of good y-bounds for zooming in on the visibility fit 

384 # in linear-y 

385 zoom_ylim_guess = abs(vis_fit_kl[np.int(.5 * len(vis_fit_kl)):]).max() 

386 zoom_bounds = [-1.1 * zoom_ylim_guess, 1.1 * zoom_ylim_guess] 

387 ax4.set_ylim(zoom_bounds) 

388 

389 # Bin the observed (real and imaginary components of the) visibilities 

390 # for plotting 

391 for i in range(len(bin_widths)): 

392 binned_vis = UVDataBinner(baselines, vis_deproj, weights, bin_widths[i]) 

393 vis_re_kl = binned_vis.V.real * 1e3 

394 vis_im_kl = binned_vis.V.imag * 1e3 

395 vis_err_re_kl = binned_vis.error.real * 1e3 

396 vis_err_im_kl = binned_vis.error.imag * 1e3 

397 vis_fit = sol.predict_deprojected(binned_vis.uv).real * 1e3 

398 

399 # Determine the visiblity domain frank fit residuals (and RMS error) 

400 # for Real(V) 

401 resid = vis_re_kl - vis_fit 

402 if norm_residuals: 

403 resid /= vis_re_kl 

404 rmse = (np.mean(resid**2))**.5 

405 

406 # Plot the observed, binned visibilities (with errorbars) and the residuals 

407 plot_vis_quantity(binned_vis.uv, vis_re_kl, ax3, c=cs[i], 

408 marker=ms[i], ls='None', 

409 label=r'Obs., {:.0f} k$\lambda$ bins'.format(bin_widths[i]/1e3)) 

410 

411 plot_vis_quantity(binned_vis.uv, vis_re_kl, ax4, c=cs[i], 

412 marker=ms[i], ls='None', 

413 label=r'Obs., {:.0f} k$\lambda$ bins'.format(bin_widths[i]/1e3)) 

414 

415 plot_vis_quantity(binned_vis.uv, vis_re_kl, ax6, c=cs[i], 

416 marker=ms[i], ls='None', 

417 label=r'Obs.>0, {:.0f} k$\lambda$ bins'.format(bin_widths[i]/1e3)) 

418 

419 plot_vis_quantity(binned_vis.uv, -vis_re_kl, ax6, c=cs2[i], 

420 marker=ms[i], ls='None', 

421 label=r'Obs.<0, {:.0f} k$\lambda$ bins'.format(bin_widths[i]/1e3)) 

422 

423 plot_vis_quantity(binned_vis.uv, vis_im_kl, ax9, c=cs[i], 

424 marker=ms[i], ls='None', 

425 label=r'Obs., {:.0f} k$\lambda$ bins'.format(bin_widths[i]/1e3)) 

426 

427 plot_vis_quantity(binned_vis.uv, resid, ax5, c=cs[i], marker=ms[i], ls='None', 

428 label=r'{:.0f} k$\lambda$ bins, RMSE {:.3f} mJy'.format(bin_widths[i]/1e3, rmse)) 

429 

430 # Plot a histogram of the observed visibilties to examine how the 

431 # visibility count varies with baseline 

432 plot_vis_hist(binned_vis, ax8, color=hist_cs[i], 

433 label=r'Obs., {:.0f} k$\lambda$ bins'.format(bin_widths[i]/1e3)) 

434 

435 # Plot the visibility domain frank fit in log-y 

436 plot_vis_quantity(grid, vis_fit_kl, ax3, c='r', label='frank') 

437 plot_vis_quantity(grid, vis_fit_kl, ax4, c='r', label='frank') 

438 plot_vis_quantity(grid, vis_fit_kl, ax6, c='r', label='frank>0') 

439 plot_vis_quantity(grid, -vis_fit_kl, ax6, c='#1EFEDC', label='frank<0') 

440 

441 # Plot the frank inferred power spectrum 

442 plot_vis_quantity(sol.q, sol.power_spectrum, ax7, label=r'$\alpha$ {:.2f}'.format( 

443 alpha) + '\n' + '$w_{smooth}$' + ' {:.1e}'.format(wsmooth)) 

444 

445 # Plot a sweep over 2\pi of the frank 1D fit 

446 # (analogous to a model image of the source) 

447 vmin = 0 

448 vmax = sol.mean.max() 

449 norm = PowerNorm(gamma, vmin, vmax) 

450 plot_2dsweep(sol.r, sol.mean, ax=ax2, cmap='inferno', norm=norm, vmin=0, vmax=vmax / 1e10) 

451 

452 ax1.set_xlabel('r ["]') 

453 ax0.set_ylabel(r'Brightness [$10^{10}$ Jy sr$^{-1}$]') 

454 ax1.set_ylabel(r'Brightness [$10^{10}$ Jy sr$^{-1}$]') 

455 ax1.set_yscale('log') 

456 ax1.set_ylim(bottom=1e-3) 

457 

458 ax2.set_xlabel('RA offset ["]') 

459 ax2.set_ylabel('Dec offset ["]') 

460 

461 ax3.set_ylabel('Re(V) [mJy]') 

462 ax4.set_ylabel('Re(V) [mJy]') 

463 if norm_residuals: 

464 ax5.set_ylabel('Norm. residual') 

465 else: 

466 ax5.set_ylabel('Residual [mJy]') 

467 ax5.set_xlabel(r'Baseline [$\lambda$]') 

468 ax3.set_xscale('log') 

469 ax4.set_xscale('log') 

470 ax5.set_xscale('log') 

471 

472 ax6.set_ylabel('Re(V) [mJy]') 

473 ax7.set_ylabel(r'Power [Jy$^2$]') 

474 ax8.set_ylabel('Count') 

475 ax9.set_ylabel('Im(V) [mJy]') 

476 ax9.set_xlabel(r'Baseline [$\lambda$]') 

477 ax6.set_xscale('log') 

478 ax6.set_yscale('log') 

479 ax7.set_xscale('log') 

480 ax7.set_yscale('log') 

481 ax8.set_xscale('log') 

482 ax8.set_yscale('log') 

483 ax9.set_xscale('log') 

484 ax6.set_ylim(bottom=1e-4) 

485 

486 xlims = ax3.get_xlim() 

487 ax4.set_xlim(xlims) 

488 ax5.set_xlim(xlims) 

489 ax6.set_xlim(xlims) 

490 ax7.set_xlim(xlims) 

491 ax8.set_xlim(xlims) 

492 ax9.set_xlim(xlims) 

493 

494 plt.setp(ax0.get_xticklabels(), visible=False) 

495 plt.setp(ax3.get_xticklabels(), visible=False) 

496 plt.setp(ax4.get_xticklabels(), visible=False) 

497 plt.setp(ax6.get_xticklabels(), visible=False) 

498 plt.setp(ax7.get_xticklabels(), visible=False)

510def make_diag_fig(r, q, iteration_diagnostics, iter_plot_range=None, 

511 force_style=True, save_prefix=None): 

512 r""" 

513 Produce a diagnostic figure showing fit convergence metrics 

514 

515 Parameters 

516 ---------- 

517 r : array 

518 Radial data coordinates at which the brightness profile is defined. 

519 The assumed unit (for the x-label) is arcsec 

520 profile_iter : list, shape = (n_iter, N_coll) 

521 Brightness profile reconstruction at each of n_iter iterations. The 

522 assumed unit (for the y-label) is Jy / sr 

523 q : array 

524 Baselines at which the power spectrum is defined. 

525 The assumed unit (for the x-label) is :math:`\lambda` 

526 iteration_diagnostics : dict 

527 The iteration_diagnostics from FrankFitter.

644def make_clean_comparison_fig(u, v, vis, weights, sol, clean_profile,

647 ): 

648 r""" 

649 Produce a figure comparing a frank fit to a CLEAN fit, in real space by 

650 convolving the frank fit with the CLEAN beam, and in visibility space by 

651 taking the discrete Hankel transform of the CLEAN profile 

652 

653 Parameters 

654 ---------- 

655 u, v : array, unit = :math:`\lambda` 

656 u and v coordinates of observations 

657 vis : array, unit = Jy

659 weights : array, unit = Jy^-2

677 frank brightness profile convolved with a CLEAN beam 

678 (see utilities.convolve_profile). 

679 The assumed unit is for the x-label

683 Whether to use preconfigured matplotlib rcParams in generated figure 

684 save_prefix : string, default = None 

685 Prefix for saved figure name. If None, the figure won't be saved 

686 

687 Returns 

688 ------- 

689 fig : Matplotlib `.Figure` instance 

690 The produced figure, including the GridSpec 

691 axes : Matplotlib `~.axes.Axes` class 

692 The axes of the produced figure

770 interp = interp1d(clean_profile['r'], clean_profile['I']) 

771 regrid_I_clean = interp(sol.r) 

772 plot_2dsweep(sol.r, regrid_I_clean, ax=ax4, cmap='inferno', norm=norm, 

773 vmin=0, vmax=vmax / 1e10, xmax=sol.Rmax, plot_colorbar=True)

775 ax0.legend(loc='best') 

776 ax1.legend(loc='best')

779 ax0.set_ylabel(r'Brightness [$10^{10}$ Jy sr$^{-1}$]')

781 ax1.set_ylabel(r'Re(V) [mJy]') 

782 ax2.set_xlabel('RA offset ["]') 

783 ax3.set_xlabel('RA offset ["]') 

784 ax4.set_xlabel('RA offset ["]') 

785 ax2.set_ylabel('Dec offset ["]')

787 ax0.set_xlim(right=sol.Rmax) 

788 if dist: 

789 xlims = ax0.get_xlim() 

790 ax0_5.set_xlim(np.multiply(xlims, dist)) 

791 

792 ax1.set_xlim(.9 * baselines.min(), 1.2 * baselines.max()) 

793 ax1.set_xscale('log') 

794 ax1.set_yscale('log') 

795 ax1.set_ylim(bottom=1e-3) 

796 

797 ax2.set_title('Unconvolved frank profile swept') 

798 ax3.set_title('Convolved frank profile swept') 

799 ax4.set_title('CLEAN profile swept') 

800 

801 ax0.text(.5, .9, 'a)', transform=ax0.transAxes) 

802 ax1.text(.5, .9, 'b)', transform=ax1.transAxes) 

803 ax2.text(.1, .9, 'c)', c='w', transform=ax2.transAxes) 

804 ax3.text(.1, .9, 'd)', c='w', transform=ax3.transAxes)
