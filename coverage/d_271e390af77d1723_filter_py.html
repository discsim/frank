26class CriticalFilter: 

27 """Optimizer for power-spectrum priors. 

28 

29 Parameters 

30 ---------- 

31 DHT : DiscreteHankelTransform 

32 A DHT object with N bins that defines H(p). The DHT is used to compute 

33 :math:`S(p)` 

34 alpha : float >= 1 

35 Order parameter of the inverse gamma prior for the power spectrum 

36 coefficients. 

37 p_0 : float >= 0, default = None, unit=Jy^2 

38 Scale parameter of the inverse gamma prior for the power spectrum 

39 coefficients. 

40 weights_smooth : float >= 0 

41 Spectral smoothness prior parameter. Zero is no smoothness prior 

42 tol : float > 0, default = 1e-3 

43 Tolerence for convergence of the power spectrum iteration. 

44 """ 

45 

46 def __init__(self, DHT, alpha, p_0, weights_smooth, 

47 tol=1e-3): 

48 

49 self._DHT = DHT 

50 

51 self._alpha = alpha 

52 self._p_0 = p_0 

53 self._rho = 1.0 

54 

55 self._tol = tol 

56 

57 self._Tij = weights_smooth * self._build_smoothing_matrix() 

58 

59 def _build_smoothing_matrix(self): 

60 log_q = np.log(self._DHT.q) 

61 dc = (log_q[2:] - log_q[:-2]) / 2 

62 de = np.diff(log_q) 

63 

64 N = self._DHT.size 

65 

66 Delta = np.zeros([3, N]) 

67 Delta[0, :-2] = 1 / (dc * de[:-1]) 

68 Delta[1, 1:-1] = - (1 / de[1:] + 1 / de[:-1]) / dc 

69 Delta[2, 2:] = 1 / (dc * de[1:]) 

70 

71 Delta = scipy.sparse.dia_matrix((Delta, [-1, 0, 1]), 

72 shape=(N, N)) 

73 

74 dce = np.zeros_like(log_q) 

75 dce[1:-1] = dc 

76 dce = scipy.sparse.dia_matrix((dce.reshape(1, -1), 0), 

77 shape=(N, N)) 

78 

79 Tij = Delta.T.dot(dce.dot(Delta)) 

80 

81 return Tij

83 def update_power_spectrum_multiple(self, fit, binning=None): 

84 """Estimate the best fit power spectrum given the current model fit. 

85 This version works when there are multiple fields being fit simultaneously. 

86 """ 

87 

88 Nfields, Np = fit.power_spectrum.shape 

89 

90 if binning is None: 

91 binning = np.zeros(Nfields, dtype='i4') 

92 Nbins = binning.max()+1 

93 

94 Ykm = self._DHT.coefficients() 

95 Tij_pI = self._Tij + scipy.sparse.identity(self._DHT.size) 

96 

97 ds = fit.MAP 

98 

99 # Project mu to Fourier-space 

100 # Tr1 = Trace(mu mu_T . Ykm_T Ykm) = Trace( Ykm mu . (Ykm mu)^T) 

101 # = (Ykm mu)**2 

102 Tr1 = np.zeros([Nbins, Np]) 

103 Y = np.zeros([Np*Nfields, Np*Nfields]) 

104 for f in range(Nfields): 

105 s = f*Np 

106 e = s+Np 

107 

108 Tr1[binning[f]] += np.dot(Ykm, ds[f]) ** 2 

109 Y[s:e, s:e] = Ykm 

110 # Project D to Fourier-space 

111 # Drr^-1 = Ykm^T Dqq^-1 Ykm 

112 # Drr = Ykm^-1 Dqq Ykm^-T 

113 # Dqq = Ykm Drr Ykm^T 

114 # Tr2 = Trace(Dqq) 

115 tmp = np.einsum('ij,ji->i', Y, fit.Dsolve(Y.T)).reshape(Nfields, Np) 

116 

117 Tr2 = np.zeros([Nbins, Np]) 

118 ps = np.zeros([Nbins, Np]) 

119 count = np.zeros(Nbins) 

120 for f in range(Nfields): 

121 Tr2[binning[f]] += tmp[f] 

122 ps[binning[f]] += fit.power_spectrum[f] 

123 count[binning[f]] += 1 

124 ps /= count.reshape(Nbins, 1) 

125 

126 for i in range(Nbins): 

127 beta = (self._p_0 + 0.5 * (Tr1[i] + Tr2[i])) / ps[i] - \ 

128 (self._alpha - 1.0 + 0.5 * self._rho * count[i]) 

129 

130 tau = scipy.sparse.linalg.spsolve(Tij_pI, beta + np.log(ps[i])) 

131 ps[i] = np.exp(tau) 

132 

133 p = np.empty_like(fit.power_spectrum) 

134 for f in range(Nfields): 

135 p[f] = ps[binning[f]] 

136 

137 return p

140 def update_power_spectrum(self, fit): 

141 """Estimate the best fit power spectrum given the current model fit""" 

142 Ykm = self._DHT.coefficients() 

143 Tij_pI = self._Tij + scipy.sparse.identity(self._DHT.size) 

144 

145 # Project mu to Fourier-space 

146 # Tr1 = Trace(mu mu_T . Ykm_T Ykm) = Trace( Ykm mu . (Ykm mu)^T) 

147 # = (Ykm mu)**2 

148 Tr1 = np.dot(Ykm, fit.MAP) ** 2 

149 # Project D to Fourier-space 

150 # Drr^-1 = Ykm^T Dqq^-1 Ykm 

151 # Drr = Ykm^-1 Dqq Ykm^-T 

152 # Dqq = Ykm Drr Ykm^T 

153 # Tr2 = Trace(Dqq) 

154 Tr2 = np.einsum('ij,ji->i', Ykm, fit.Dsolve(Ykm.T)) 

155 

156 pi = fit.power_spectrum 

157 

158 beta = (self._p_0 + 0.5 * (Tr1 + Tr2)) / pi - \ 

159 (self._alpha - 1.0 + 0.5 * self._rho) 

160 

161 tau = scipy.sparse.linalg.spsolve(Tij_pI, beta + np.log(pi)) 

162 

163 return np.exp(tau) 

164 

165 def check_convergence(self, pi_new, pi_old): 

166 """Determine whether the power-spectrum iterations have converged""" 

167 return np.all(np.abs(pi_new - pi_old) <= self._tol * pi_new)

170 def covariance_MAP(self, fit): 

171 """ 

172 Covariance of the power spectrum at maximum likelihood 

173 

174 Parameters 

175 ---------- 

176 fit : _HankelRegressor 

177 Solution at maximum likelihood 

178 

179 Returns 

180 ------- 

181 ps_cov : 2D array 

182 Covariance matrix of the power spectrum at maximum likelihood 

183 

184 Notes 

185 ----- 

186 Only valid at the location of maximum likelihood 

187 

188 """ 

189 Ykm = self._DHT.coefficients() 

190 

191 mq = np.dot(Ykm, fit.MAP) 

192 

193 mqq = np.outer(mq, mq) 

194 Dqq = np.dot(Ykm, fit.Dsolve(Ykm.T)) 

195 

196 p = fit.power_spectrum 

197 tau = np.log(p) 

198 

199 hess = \ 

200 + np.diag(self._alpha - 1.0 + 0.5 * self._rho + self._Tij.dot(tau)) \ 

201 + self._Tij.todense() \ 

202 - 0.5 * np.outer(1 / p, 1 / p) * (2 * mqq + Dqq) * Dqq 

203 

204 # Invert the Hessian 

205 hess_chol = scipy.linalg.cho_factor(hess) 

206 ps_cov = scipy.linalg.cho_solve(hess_chol, np.eye(self._DHT.size)) 

207 

208 return ps_cov

210 def log_prior(self, p): 

211 """ 

212 Compute the log prior probability, log(P(p)), 

213 

214 .. math: 

215 `log[P(p)] ~ np.sum(p0/pi - alpha*np.log(p0/pi)) 

216 - 0.5*np.log(p) (weights_smooth*T) np.log(p)` 

217 

218 Parameters 

219 ---------- 

220 p : array, size = N 

221 Power spectrum coefficients. 

222  

223 Returns 

224 ------- 

225 log[P(p)] : float 

226 Log prior probability 

227 

228 Notes 

229 ----- 

230 Computed up to a normalizing constant that depends on alpha and p0 

231 """ 

232 

233 # Add the power spectrum prior term 

234 xi = self._p_0 / p 

235 like = np.sum(xi - self._alpha * np.log(xi)) 

236 

237 # Extra term due to spectral smoothness 

238 tau = np.log(p) 

239 like -= 0.5 * np.dot(tau, self._Tij.dot(tau)) 

240 

241 return like
