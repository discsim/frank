19"""This module has functions that read in a UVTable and save fit results to 

20 file. 

21""" 

22 

23import os 

24import numpy as np 

25import pickle 

26import logging 

27 

28def load_uvtable(data_file): 

29 r""" 

30 Read in a UVTable with data to be fit 

31 

32 Parameters 

33 ---------- 

34 data_file : string 

35 UVTable with data to be fit, with columns: 

36 u [lambda] v [lambda] Re(V) [Jy] Im(V) [Jy] Weight [Jy^-2] 

37 

38 Returns 

39 ------- 

40 u, v : array, unit = :math:`\lambda` 

41 u and v coordinates of observations 

42 vis : array, unit = Jy 

43 Observed visibilities (complex: real + imag * 1j) 

44 weights : array, unit = Jy^-2 

45 Weights on the visibilities, of the form 

46 :math:`1 / \sigma^2` 

47 """ 

48 

49 logging.info(' Loading UVTable') 

50 

51 # Get extension removing compressed part 

52 base, extension = os.path.splitext(data_file) 

53 if extension in {'.gz', '.bz2'}: 

54 extension = os.path.splitext(base)[1] 

55 if extension not in {'.txt', '.dat'}: 

56 raise ValueError("Compressed UV tables (`.gz` or `.bz2`) must be in " 

57 "one of the formats `.txt` or `.dat`.") 

58 

59 if extension in {'.txt', '.dat'}: 

60 u, v, re, im, weights = np.genfromtxt(data_file).T 

61 vis = re + 1j*im 

62 

63 elif extension == '.npz': 

64 dat = np.load(data_file) 

65 u, v, vis, weights = [dat[i] for i in ['u', 'v', 'V', 'weights']] 

66 if not np.iscomplexobj(vis): 

67 raise ValueError("You provided a UVTable with the extension {}." 

68 " This extension requires the UVTable's variable 'V' to be" 

69 " complex (of the form Re(V) + Im(V) * 1j).".format(extension)) 

70 

71 else: 

72 raise ValueError("You provided a UVTable with the extension {}." 

73 " Please provide it as a `.txt`, `.dat`, `.npy`, or" 

74 " `.npz`. Formats .txt and .dat may optionally be" 

75 " compressed (`.gz`, `.bz2`).".format(extension)) 

76 

77 return u, v, vis, weights

80def save_uvtable(filename, u, v, vis, weights): 

81 r"""Save a uvtable to file. 

82 

83 Parameters 

84 ---------- 

85 filename : string 

86 File to save the uvtable to. 

87 u, v : array, unit = :math:`\lambda` 

88 u and v coordinates of observations 

89 vis : array, unit = Jy 

90 Complex visibilities 

91 weights : array, unit = Jy^-2 

92 Weights on the visibilities, of the form 

93 :math:`1 / \sigma^2` 

94 """ 

95 

96 extension = os.path.splitext(filename)[1] 

97 if extension not in {'.txt', '.dat', '.npz'}: 

98 raise ValueError("file extension must be 'npz', 'txt', or 'dat'.") 

99 

100 if extension in {'.txt', '.dat'}: 

101 header = 'u [lambda]\tv [lambda]\tRe(V) [Jy]\tIm(V) [Jy]\tWeight [Jy^-2]' 

102 

103 np.savetxt(filename, 

104 np.stack([u, v, vis.real, vis.imag, 

105 weights], axis=-1), 

106 header=header) 

107 

108 elif extension == '.npz': 

109 np.savez(filename, 

110 u=u, v=v, V=vis, weights=weights, 

111 units={'u': 'lambda', 'v': 'lambda', 

112 'V': 'Jy', 'weights': "Jy^-2"})

115def save_fit(u, v, vis, weights, sol, prefix, save_solution=True, 

116 save_profile_fit=True, save_vis_fit=True, save_uvtables=True, 

117 save_iteration_diag=False, iteration_diag=None, 

118 format='npz', 

119 ): 

120 r""" 

121 Save datafiles of fit results 

122 

123 Parameters 

124 ---------- 

125 u, v : array, unit = :math:`\lambda` 

126 u and v coordinates of observations 

127 vis : array, unit = Jy 

128 Complex visibilities 

129 weights : array, unit = Jy^-2 

130 Weights on the visibilities, of the form 

131 :math:`1 / \sigma^2` 

132 sol : _HankelRegressor object 

133 Reconstructed profile using Maximum a posteriori power spectrum 

134 (see frank.radial_fitters.FrankFitter) 

135 prefix : string 

136 Base part of the filename to which files will be saved 

137 save_solution : bool 

138 Whether to save `sol` object (see frank.radial_fitters.FrankFitter) 

139 save_profile_fit : bool 

140 Whether to save fitted brightness profile 

141 save_vis_fit : bool 

142 Whether to save fitted visibility distribution 

143 NOTE: This is deprojected 

144 save_uvtables : bool 

145 Whether to save fitted and residual UVTables 

146 NOTE: These are reprojected 

147 save_iteration_diag : bool 

148 Whether to save diagnostics of the fit iteration 

149 iteration_diag : dict 

150 Diagnostics of the fit iteration 

151 (see radial_fitters.FrankFitter.fit) 

152 format : string, default = 'npz' 

153 File format in which to save the fit's output UVTable(s) 

154 """ 

155 

156 logging.info(' Saving fit results to {}*'.format(prefix)) 

157 

158 

159 if not format in {'txt', 'dat', 'npz'}: 

160 raise ValueError("'format' must be 'npz', 'txt', or 'dat'.") 

161 

162 if save_solution: 

163 with open(prefix + '_frank_sol.obj', 'wb') as f: 

164 pickle.dump(sol, f) 

165 

166 if save_iteration_diag: 

167 with open(prefix + '_frank_iteration_diagnostics.obj', 'wb') as f: 

168 pickle.dump(iteration_diag, f) 

169 

170 if save_profile_fit: 

171 np.savetxt(prefix + '_frank_profile_fit.txt', 

172 np.array([sol.r, sol.mean, np.diag(sol.covariance)**.5]).T, 

173 header='r [arcsec]\tI [Jy/sr]\tI_uncer [Jy/sr]') 

174 

175 if save_vis_fit: 

176 np.savetxt(prefix + '_frank_vis_fit.' + format, 

177 np.array([sol.q, sol.predict_deprojected(sol.q).real]).T, 

178 header='Baseline [lambda]\tProjected Re(V) [Jy]') 

179 

180 

181 if save_uvtables: 

182 logging.info(' Saving fit and residual UVTables. N.B.: These will' 

183 ' be of comparable size to your input UVTable') 

184 

185 V_pred = sol.predict(u, v) 

186 

187 save_uvtable(prefix + '_frank_uv_fit.' + format, 

188 u, v, V_pred, weights) 

189 save_uvtable(prefix + '_frank_uv_resid.' + format, 

190 u, v, vis - V_pred, weights)
