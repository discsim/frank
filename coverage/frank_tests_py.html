34def test_hankel_gauss(): 

35 """Check the Hankel transform""" 

36 

37 def gauss_real_space(r): 

38 x = r 

39 return np.exp(-0.5 * x * x) 

40 

41 def gauss_vis_space(q): 

42 qs = (2 * np.pi) * q 

43 return np.exp(-0.5 * qs * qs) * (2 * np.pi) 

44 

45 DHT = DiscreteHankelTransform(5.0, 100) 

46 

47 Ir = gauss_real_space(DHT.r) 

48 Iq = gauss_vis_space(DHT.q) 

49 

50 # Test at the DHT points. 

51 # Use a large error estimate because the DHT is approximate 

52 np.testing.assert_allclose(Iq, DHT.transform(Ir, direction='forward'), 

53 atol=1e-5, rtol=0, err_msg="Forward DHT") 

54 

55 np.testing.assert_allclose(Ir, DHT.transform(Iq, direction='backward'), 

56 atol=1e-5, rtol=0, err_msg="Inverse DHT") 

57 

58 # Test at generic points. 

59 # Larger error needed 

60 q = np.linspace(0.0, 1.0, 25) 

61 np.testing.assert_allclose(gauss_vis_space(q), 

62 DHT.transform(Ir, q=q, direction='forward'), 

63 atol=1e-4, rtol=0, err_msg="Generic Forward DHT") 

64 

65 r = np.linspace(0, 5.0, 25) 

66 np.testing.assert_allclose(gauss_real_space(r), 

67 DHT.transform(Iq, q=r, direction='backward'), 

68 atol=1e-4, rtol=0, err_msg="Generic Inverse DHT") 

69 

70 # Check the coefficients matrix works 

71 Hf = DHT.coefficients(direction='forward') 

72 Hb = DHT.coefficients(direction='backward') 

73 

74 np.testing.assert_allclose(np.dot(Hf, Ir), 

75 DHT.transform(Ir, direction='forward'), 

76 rtol=1e-7, err_msg="Forward DHT Coeffs") 

77 np.testing.assert_allclose(np.dot(Hb, Iq), 

78 DHT.transform(Iq, direction='backward'), 

79 rtol=1e-7, err_msg="Inverse DHT Coeffs") 

80 

81 # Compare cached vs non-cached DHT points 

82 np.testing.assert_allclose(DHT.coefficients(q=DHT.q), 

83 DHT.coefficients(), 

84 atol=1e-12, rtol=0, 

85 err_msg="Cached forward DHT Coeffs" 

86 ) 

87 np.testing.assert_allclose(DHT.coefficients(q=DHT.r, direction='backward'), 

88 DHT.coefficients(direction='backward'), 

89 atol=1e-12, rtol=0, 

90 err_msg="Cached inverse DHT Coeffs" 

91 )

99def load_AS209(uv_cut=None): 

100 """Load data for subsequent tests""" 

101 uv_AS209_DSHARP = np.load('docs/tutorials/AS209_continuum.npz') 

102 geometry = FixedGeometry(dRA=-1.9e-3, dDec=2.5e-3, inc=34.97, 

103 PA=85.76) 

104 

105 if uv_cut is not None: 

106 u, v = [uv_AS209_DSHARP[x] for x in ['u', 'v']] 

107 

108 q = np.hypot(*geometry.deproject(u,v)) 

109 

110 keep = q < uv_cut 

111 

112 cut_data = {} 

113 for key in uv_AS209_DSHARP: 

114 if key not in { 'u', 'v', 'V', 'weights' }: 

115 continue 

116 cut_data[key] = uv_AS209_DSHARP[key][keep] 

117 

118 uv_AS209_DSHARP = cut_data 

119 

120 return uv_AS209_DSHARP, geometry

232def test_uvbin(): 

233 """Check the uv-data binning routine""" 

234 AS209, geometry = load_AS209() 

235 

236 uv = np.hypot(*geometry.deproject(AS209['u'], AS209['v'])) 

237 

238 uvbin = UVDataBinner(uv, AS209['V'], AS209['weights'], 50e3) 

239 

240 uvmin = 1e6 

241 uvmax = 1e6 + 50e3 

242 

243 idx = (uv >= uvmin) & (uv < uvmax) 

244 

245 widx = AS209['weights'][idx] 

246 

247 w = np.sum(widx) 

248 V = np.sum(widx*AS209['V'][idx]) / w 

249 q = np.sum(widx*uv[idx]) / w 

250 

251 i = (uvbin.uv >= uvmin) & (uvbin.uv < uvmax) 

252 

253 np.testing.assert_allclose(q, uvbin.uv[i]) 

254 np.testing.assert_allclose(V, uvbin.V[i]) 

255 np.testing.assert_allclose(w, uvbin.weights[i]) 

256 np.testing.assert_allclose(len(widx), uvbin.bin_counts[i])

259def _run_pipeline(geometry='gaussian', fit_phase_offset=True, make_figs=False, 

260 multifit=False, bootstrap=False): 

261 """Check the full pipeline that performs a fit and outputs results""" 

262 

263 # Build a subset of the data that we'll load during the fit 

264 AS209, AS209_geometry = load_AS209(uv_cut=1e6) 

265 

266 u, v, vis, weights = [AS209[k][::100] for k in ['u', 'v', 'V', 'weights']] 

267 

268 tmp_dir = '/tmp/frank/tests' 

269 os.makedirs(tmp_dir, exist_ok=True) 

270 

271 uv_table = os.path.join(tmp_dir, 'small_uv.npz') 

272 save_uvtable(uv_table, u, v, vis, weights) 

273 

274 # Build a parameter file 

275 params = fit.load_default_parameters() 

276 

277 params['input_output']['uvtable_filename'] = uv_table 

278 

279 # Set the model parameters 

280 params['hyperparameters']['n'] = 20 

281 params['hyperparameters']['rout'] = 1.6 

282 params['hyperparameters']['alpha'] = 1.05 

283 params['hyperparameters']['wmsooth'] = 1e-2 

284 

285 geom = params['geometry'] 

286 geom['type'] = geometry 

287 geom['fit_phase_offset'] = fit_phase_offset 

288 geom['inc'] = AS209_geometry.inc 

289 geom['pa'] = AS209_geometry.PA 

290 geom['dra'] = AS209_geometry.dRA 

291 geom['ddec'] = AS209_geometry.dDec 

292 

293 if make_figs: 

294 params['plotting']['quick_plot'] = True 

295 params['plotting']['full_plot'] = True 

296 params['plotting']['diag_plot'] = True 

297 params['plotting']['deprojec_plot'] = True 

298 params['plotting']['save_figs'] = True 

299 params['plotting']['distance'] = 121. 

300 params['plotting']['bin_widths'] = [1e5] 

301 params['plotting']['iter_plot_range'] = [0, 5] 

302 params['analysis']['compare_profile'] = 'docs/tutorials/AS209_clean_profile.txt' 

303 params['analysis']['clean_beam'] = {'bmaj' : 0.03883, 

304 'bmin' : 0.03818, 

305 'beam_pa' : 85.82243 

306 } 

307 

308 else: 

309 params['plotting']['quick_plot'] = False 

310 params['plotting']['full_plot'] = False 

311 params['plotting']['diag_plot'] = False 

312 params['plotting']['deprojec_plot'] = False 

313 

314 if multifit: 

315 params['hyperparameters']['alpha'] = [1.05, 1.30] 

316 

317 if bootstrap: 

318 params['analysis']['bootstrap_ntrials'] = 2 

319 

320 # Save the new parameter file 

321 param_file = os.path.join(tmp_dir, 'params.json') 

322 with open(param_file, 'w') as f: 

323 json.dump(params, f) 

324 

325 # Call the pipeline to perform the fit 

326 fit.main(['-p', param_file])
