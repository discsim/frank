36def test_hankel_gauss(): 

37 """Check the Hankel transform""" 

38 

39 def gauss_real_space(r): 

40 x = r 

41 return np.exp(-0.5 * x * x) 

42 

43 def gauss_vis_space(q): 

44 qs = (2 * np.pi) * q 

45 return np.exp(-0.5 * qs * qs) * (2 * np.pi) 

46 

47 DHT = DiscreteHankelTransform(5.0, 100) 

48 

49 Ir = gauss_real_space(DHT.r) 

50 Iq = gauss_vis_space(DHT.q) 

51 

52 # Test at the DHT points. 

53 # Use a large error estimate because the DHT is approximate 

54 np.testing.assert_allclose(Iq, DHT.transform(Ir, direction='forward'), 

55 atol=1e-5, rtol=0, err_msg="Forward DHT") 

56 

57 np.testing.assert_allclose(Ir, DHT.transform(Iq, direction='backward'), 

58 atol=1e-5, rtol=0, err_msg="Inverse DHT") 

59 

60 # Test at generic points. 

61 # Larger error needed 

62 q = np.linspace(0.0, 1.0, 25) 

63 np.testing.assert_allclose(gauss_vis_space(q), 

64 DHT.transform(Ir, q=q, direction='forward'), 

65 atol=1e-4, rtol=0, err_msg="Generic Forward DHT") 

66 

67 r = np.linspace(0, 5.0, 25) 

68 np.testing.assert_allclose(gauss_real_space(r), 

69 DHT.transform(Iq, q=r, direction='backward'), 

70 atol=1e-4, rtol=0, err_msg="Generic Inverse DHT") 

71 

72 # Check the coefficients matrix works 

73 Hf = DHT.coefficients(direction='forward') 

74 Hb = DHT.coefficients(direction='backward') 

75 

76 np.testing.assert_allclose(np.dot(Hf, Ir), 

77 DHT.transform(Ir, direction='forward'), 

78 rtol=1e-7, err_msg="Forward DHT Coeffs") 

79 np.testing.assert_allclose(np.dot(Hb, Iq), 

80 DHT.transform(Iq, direction='backward'), 

81 rtol=1e-7, err_msg="Inverse DHT Coeffs") 

82 

83 # Compare cached vs non-cached DHT points 

84 np.testing.assert_allclose(DHT.coefficients(q=DHT.q), 

85 DHT.coefficients(), 

86 atol=1e-12, rtol=0, 

87 err_msg="Cached forward DHT Coeffs" 

88 ) 

89 np.testing.assert_allclose(DHT.coefficients(q=DHT.r, direction='backward'), 

90 DHT.coefficients(direction='backward'), 

91 atol=1e-12, rtol=0, 

92 err_msg="Cached inverse DHT Coeffs" 

93 )

101def load_AS209(uv_cut=None): 

102 """Load data for subsequent tests""" 

103 uv_AS209_DSHARP = np.load('docs/tutorials/AS209_continuum.npz') 

104 geometry = FixedGeometry(dRA=-1.9e-3, dDec=2.5e-3, inc=34.97, 

105 PA=85.76) 

106 

107 if uv_cut is not None: 

108 u, v = [uv_AS209_DSHARP[x] for x in ['u', 'v']] 

109 

110 q = np.hypot(*geometry.deproject(u,v)) 

111 

112 keep = q < uv_cut 

113 

114 cut_data = {} 

115 for key in uv_AS209_DSHARP: 

116 if key not in { 'u', 'v', 'V', 'weights' }: 

117 continue 

118 cut_data[key] = uv_AS209_DSHARP[key][keep] 

119 

120 uv_AS209_DSHARP = cut_data 

121 

122 return uv_AS209_DSHARP, geometry

297def test_uvbin(): 

298 """Check the uv-data binning routine""" 

299 AS209, geometry = load_AS209() 

300 

301 uv = np.hypot(*geometry.deproject(AS209['u'], AS209['v'])) 

302 

303 uvbin = UVDataBinner(uv, AS209['V'], AS209['weights'], 50e3) 

304 

305 uvmin = 1e6 

306 uvmax = 1e6 + 50e3 

307 

308 idx = (uv >= uvmin) & (uv < uvmax) 

309 

310 widx = AS209['weights'][idx] 

311 

312 w = np.sum(widx) 

313 V = np.sum(widx*AS209['V'][idx]) / w 

314 q = np.sum(widx*uv[idx]) / w 

315 

316 i = (uvbin.uv >= uvmin) & (uvbin.uv < uvmax) 

317 

318 np.testing.assert_allclose(q, uvbin.uv[i]) 

319 np.testing.assert_allclose(V, uvbin.V[i]) 

320 np.testing.assert_allclose(w, uvbin.weights[i]) 

321 np.testing.assert_allclose(len(widx), uvbin.bin_counts[i])

324def _run_pipeline(geometry='gaussian', fit_phase_offset=True, 

325 fit_inc_pa=True, make_figs=False, 

326 multifit=False, bootstrap=False): 

327 """Check the full pipeline that performs a fit and outputs results""" 

328 

329 # Build a subset of the data that we'll load during the fit 

330 AS209, AS209_geometry = load_AS209(uv_cut=1e6) 

331 

332 u, v, vis, weights = [AS209[k][::100] for k in ['u', 'v', 'V', 'weights']] 

333 

334 tmp_dir = '/tmp/frank/tests' 

335 os.makedirs(tmp_dir, exist_ok=True) 

336 

337 uv_table = os.path.join(tmp_dir, 'small_uv.npz') 

338 save_uvtable(uv_table, u, v, vis, weights) 

339 

340 # Build a parameter file 

341 params = fit.load_default_parameters() 

342 

343 params['input_output']['uvtable_filename'] = uv_table 

344 

345 # Set the model parameters 

346 params['hyperparameters']['n'] = 20 

347 params['hyperparameters']['rout'] = 1.6 

348 params['hyperparameters']['alpha'] = 1.05 

349 params['hyperparameters']['wmsooth'] = 1e-2 

350 

351 geom = params['geometry'] 

352 geom['type'] = geometry 

353 geom['fit_phase_offset'] = fit_phase_offset 

354 geom['fit_inc_pa'] = fit_inc_pa 

355 geom['inc'] = AS209_geometry.inc 

356 geom['pa'] = AS209_geometry.PA 

357 geom['dra'] = AS209_geometry.dRA 

358 geom['ddec'] = AS209_geometry.dDec 

359 

360 if make_figs: 

361 params['plotting']['quick_plot'] = True 

362 params['plotting']['full_plot'] = True 

363 params['plotting']['diag_plot'] = True 

364 params['plotting']['deprojec_plot'] = True 

365 params['plotting']['save_figs'] = True 

366 params['plotting']['distance'] = 121. 

367 params['plotting']['bin_widths'] = [1e5] 

368 params['plotting']['iter_plot_range'] = [0, 5] 

369 params['analysis']['compare_profile'] = 'docs/tutorials/AS209_clean_profile.txt' 

370 params['analysis']['clean_beam'] = {'bmaj' : 0.03883, 

371 'bmin' : 0.03818, 

372 'beam_pa' : 85.82243 

373 } 

374 

375 else: 

376 params['plotting']['quick_plot'] = False 

377 params['plotting']['full_plot'] = False 

378 params['plotting']['diag_plot'] = False 

379 params['plotting']['deprojec_plot'] = False 

380 

381 if multifit: 

382 params['hyperparameters']['alpha'] = [1.05, 1.30] 

383 

384 if bootstrap: 

385 params['analysis']['bootstrap_ntrials'] = 2 

386 

387 # Save the new parameter file 

388 param_file = os.path.join(tmp_dir, 'params.json') 

389 with open(param_file, 'w') as f: 

390 json.dump(params, f) 

391 

392 # Call the pipeline to perform the fit 

393 fit.main(['-p', param_file])
